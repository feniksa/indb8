include_directories(${CMAKE_SOURCE_DIR}/inc)

# --------------------------------
# dbcore target
# ---------------------------------
set (DB_TEST_SOURCES
     MojDbTestRunner.cpp
     MojDbTestStorageEngine.cpp
)

foreach(test
        MojDbBulk
        MojDbConcurrency
        MojDbCrud
        MojDbDumpLoad
        MojDbIndex
        MojDbKind
        MojDbLocale
        MojDbPermission
        MojDbPurge
        MojDbQuery
        MojDbQueryFilter
        MojDbQuota
        MojDbRev
        MojDbRevisionSet
        MojDbSearch
        MojDbDistinct
        MojDbTextCollator
        MojDbTextTokenizer
        MojDbWatch
        MojDbWhere
        MojDbTxn
        MojDbCursorTxn
        MojDbNewId
        MojDbShardManager
        )
    add_test(NAME ${test} COMMAND test_db_core ${test})
    set(DB_TEST_SOURCES ${DB_TEST_SOURCES} ${test}Test.cpp)
endforeach()

add_executable(test_db_core ${DB_TEST_SOURCES})
target_link_libraries(test_db_core
                      mojocore
                      mojodb
                      mojodbengine                      
)
install( TARGETS test_db_core
    RUNTIME DESTINATION lib/indb8/tests
)
add_test(test_db_core test_db_core)
